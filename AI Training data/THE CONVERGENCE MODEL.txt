# ✅ UNIVERSAL NORMALIZATION HELPERS (Required)

#[inline]
fn clamp01(x: f64) -> f64 {
    if x < 0.0 { 0.0 }
    else if x > 1.0 { 1.0 }
    else { x }
}

#[inline]
fn dignity_ratio(w: f64, c: f64) -> f64 {
    clamp01(w / c)
}

#[inline]
fn debt_factor(d: f64) -> f64 {
    // Hard cap at 200% Debt/GDP → zero capacity
    let df = 1.0 - (d / 2.0);
    clamp01(df)
}

#[inline]
fn birth_momentum(tfr: f64) -> f64 {

    clamp01(tfr / 2.1)
}
